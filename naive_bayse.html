from sklearn.naive_bayes import MultinomialNB
from sklearn.feature_extraction.text import CountVectorizer
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, classification_report

# 1.     ( )
emails = [
    "!      !",  # 
    ".   .",      # 
    ":    !",          # 
    "    ?",          # 
    "[]     .", #  (   )
    "!  100 !",   # 
    "  .",                # 
    " IT   ",              # 
    " !   !",        # 
    "   ."          # 
]
labels = [1, 0, 1, 0, 0, 1, 0, 0, 1, 0]  # 1: , 0: 
# def custom_preprocessor(text):
#     text = text.lower()
#     return text.replace("document","")

# vectorizer = CountVectorizer(preprocessor=custom_preprocessor)

# 2.      (CountVectorizer)
#     .
vectorizer = CountVectorizer()
X = vectorizer.fit_transform(emails)
# print(X.toarray())
# 3.     
X_train, X_test, y_train, y_test = train_test_split(X, labels, test_size=0.2, random_state=42)

# 4.     (MultinomialNB -   )
model = MultinomialNB()
model.fit(X_train, y_train)

# 5.   
y_pred = model.predict(X_test)

# 6.   
accuracy = accuracy_score(y_test, y_pred)
report = classification_report(y_test, y_pred, target_names=['', ''])

print(":", accuracy)
print("\n :\n", report)

# 7.    
new_emails = [
    "   ",
    "! 90%  !",
    "    "
]
new_X = vectorizer.transform(new_emails)
new_predictions = model.predict(new_X)

print("\n  :")
for email, prediction in zip(new_emails, new_predictions):
    print(f": '{email}' -> : {'' if prediction == 1 else ''}")
